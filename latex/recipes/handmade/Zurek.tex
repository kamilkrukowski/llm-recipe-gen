\begin{recipe}
[
    preparationtime = {\unit[24]{h}},
    source = Kamil Krukowski,
    portion = \portion{6}
]
{White Borscht (Zurek)}

\setRecipeColors{
recipename = orange,
numeration = black
}

    \introduction{
    Rustic Polish-American holiday fare, developed and honed by families throughout the American Northeast. The tangy complexity of a soup based on a sourdough starter can't be achieved any other way. This is a perfect recipe to make-ahead and enjoy cold on a summer day or hot on a winter night.}
    
    \ingredients
    {
            \unit[1.5]{cup} & Sourdough Starter\\
            \unit[1]{lb} & Kielbasa \\
            \unit[6]{cups} & broth \\
            1 & medium sweet onion, chopped \\
            \unit[3]{cloves} &  garlic, minced\\
            \\
            \textbf{Spice Tea} & \\
            \\
            \unit[4-5] & Allspice berries \\
            \unit[2] & Bay Leaves \\
            \unit[2]{tbsp} & brown sugar\\
            \unit[1]{tbsp} & Marjoram \\
            \\
            \textbf{Garnish} & \\
            \\
            4-6 & hardboiled eggs \\
            \unit[\nicefrac{1}{4}]{cup} & dill, chopped \\
    }
    
    \preparation
    { 
        \step Bring large pot of broth to a boil, then down to a simmer with Kielbasa. Remove and chop the kielbasa in one-inch chunks when fully cooked, about 25 minutes. Set aside the broth.
        \step Combine spice tea ingredients with hot water. Set aside to let steep for at least 10 minutes. Alternatively, the spice tea can be added to the starter a day prior.
        \step Brown the chopped kielbasa over high heat. Add to the broth. Saut\'e chopped onions and garlic until fragrant. Pur\'ee the vegetables and add them to the broth.
        \step Let the broth cool to room temperature. The flavors of the sourdough broth will keep improving for a few days as the starter ferments the ingredients. This will stop once the soup is heated up to 140\textdegree f which stops all sourdough activity permanently.
        \step Introduce the sourdough starter and strained spice tea to the broth. Salt as necessary.
        \step Heat up individual portions immediately before serving. Quarter the hardboiled eggs. Garnish soup with eggs, black pepper, and chopped herbs. Enjoy!
    }
    
    \hint{
        \begin{itemize}
            \item In a time crunch, substitute sourdough starter with 1 cup sour cream and \nicefrac{1}{2} cup flour. The cooling step can be completely omitted
            \item Consider adding smoked bacon for extra deliciousness.
        \end{itemize}
    }
    


\end{recipe}
