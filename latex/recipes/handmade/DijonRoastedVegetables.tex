\begin{recipe}
[
    preparationtime = {\unit[1.5]{h}},
    source = Kamil Krukowski,
    portion = \portion{8}
]
{Honey-Dijon Pink Vegetables}

\setRecipeColors{
recipename = orange,
numeration = black
}

    \introduction{
    A family favorite inspired by slavic influence. The roast vegetables offer a refreshing hot twist on traditional Russian Vinegret salad.
    }
    
    \ingredients
    {
        5 & large carrots \\
        1 & parsnip \\
        3 & medium beets \\
        2 & tart apples \\
        \\
        \textbf{Sauce} & \\
        \\
        \unit[\nicefrac{1}{4}]{cup} & dijon mustard \\
        \unit[\nicefrac{1}{4}]{cup} & honey \\
        \unit[\nicefrac{1}{4}]{cup} & olive oil \\
        \unit[1]{tbsp} & Salt 
    }

    \preparation
    { 
        \step Peel and dice parsnips and beets into rustic cubes. Combine with oil and bake at 375f for 20 minutes.
        \step Cut carrots into thin rectangular strips. Toss carrots in mustard, honey, and salt.
        \step Combine carrots with parsnips and beets. Mix thoroughly and return to oven at 375f for 25 minutes.
    }

    \hint{

    \begin{itemize}
        \item We recommend Granny Smith apples
        \item Use gloves when handling beets
        \item If a sweeter apple is used, some apple cider vinegar or lemon can be a great finish to the dish.
        \item Thyme and Parsley are good garnishes.
        \item Consider substituting maple syrup, brown sugar glaze, or balsamic glaze for honey.
    \end{itemize}
    }
    


\end{recipe}
